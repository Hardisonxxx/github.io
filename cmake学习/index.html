安装Cmake

CMake是一个管理代码构建的工具。

linux下用apt安装g++和cmake

windows下安装编译工具：w64devkit：https://github.com/skeeto/w64devkit/releases

将上述工具添加到环境变量里：

这里提一嘴，常用的编译器有MSVC和g++等等等等，MSVC是微软的编译器，g++等是GUN的编译器（GUN其实就是开源的意思），而我们上面安装的编译工具w64devkit就是包含了g++等GUN编译器的编译工具集合，也就是所谓的用于x64 Windows的便携式C和c++开发工具包。

如果你的电脑上安装了Vs studio，在Vs studio install tools里可以选择安装MSVC编译器在你的电脑上。

指定CMake默认的编译器

在对CMakeLists.txt根目录使用cmake进行操作的时候，通常调用的是MSVC编译器进行编译，如需更换默认编译器，需要在环境变量中CMAke添加名在CMAKE_GENERATOR，值为MinGW Makefiles（值可以随意指定，想用哪个编译器就用哪个，具体在下方展示）

CMake代码目录结构

有一些Cpp文件和头文件，同时还有一个CMakeLists.txt文件

CMakeLists.txt怎么写？

开始

CMakeLists.txt其实就是一个配置文件，在CMakeLists.txt里面可以对项目进行配置

先从最简单的开始，让一个程序能够跑起来

cmake_minimum_required(VERSION 3.10)  #表示最低支持的版本
project(HelloWorld) #项目名称
add_executable(HelloWorld add.c div.c main.c mult.c sub.c) # 需要成的软件名字以及项目内需要包含的cpp文件

然后直接使用cmake命令生成Makefile文件，然后执行make命令进行编译即可。

set

在cmake里定义变量需要使用set

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)
set(SRC add.cpp div.cpp main.cpp mult.cpp sub.cpp)  #将add.cpp div.cpp main.cpp mult.cpp sub.cpp这些名称赋值给SRC变量
add_executable(HelloWorld ${SRC}) #调用SRC变量，语法为${}

指定使用C++标准

在CMakefiles.txt中用set指定：

#增加-std=c++11
set(CMAKE_CXX_STANDARD 11)
#增加-std=c++14
set(CMAKE_CXX_STANDARD 14)
#增加-std=c++17
set(CMAKE_CXX_STANDARD 17)

指定输出路径

在CMake中指定可执行程序输出的路径，也对应一个宏，叫做EXECUTABLE_OUTPUT_PATH，它的值还是通过set命令进行设置:

set(HOME /home/xxxx/Linux/xxxx) #命名将路径的值赋给变量HOME // 活学活用
set(EXECUTABLE_OUTPUT_PATH ${HOME}/bin)   #使用EXECUTABLE_OUTPUT_PATH 设置输出路径

如果这个路径中的子目录不存在，会自动生成，无需自己手动创建

搜索源文件

方法1

一个正式项目中的源文件很多，我们在写add_executable的时候也不太可能把每一个源文件都写上，所以我们需要一个语句来遍历一下源文件并添加一下。CMake为我们提供提供了一个aux_sourse_directory命令，可以直接查找一个目录下的源文件。

aux_source_directory(< dir > < variable >)

eg:

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)
# set(SRC add.cpp div.cpp main.cpp mult.cpp sub.cpp)
aux_source_directory(${PROJECT_SOURCE_DIR} SRC) #使用PROJECT_SOURCE_DIR宏和aux_source_directory命令遍历项目根文件目录，然后赋值给SRC变量
set(EXECUTABLE_OUTPUT_PATH C:/Users/Denglzzz/Desktop/code/Cmaketest/build) #设置输出文件路径
add_executable(HelloWorld ${SRC})

方法2

方法2使用file命令来查找源文件：file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)
# set(SRC add.cpp div.cpp main.cpp mult.cpp sub.cpp)
# aux_source_directory(${PROJECT_SOURCE_DIR} SRC) 
file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
set(EXECUTABLE_OUTPUT_PATH /root/Cmaketest/build) #设置输出文件路径
add_executable(HelloWorld ${SRC})

aux_source_directory(${PROJECT_SOURCE_DIR} SRC)

使用PROJECT_SOURCE_DIR宏和aux_source_directory命令遍历项目根文件目录，PROJECT_SOURCE_DIR宏是表示执行cmake命令时后面所跟随的路径，其实也就是CMakelists.txt所在的路劲。

file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)

使用CMAKE_CURRENT_SOURCE_DIR宏和file命令遍历项目根文件目录,将后缀名为.cpp文件名加入变量SRC中，CMAKE_CURRENT_SOURCE_DIR宏是表示CMakelists.txt文件所在的目录。

指定头文件目录

为了保证项目文件的整洁，通常将源文化、头文件等分别放入对应的文件夹内，具体目录树如下：

由于源文件和头文件不在一个目录，需要在CMakelists.txt文件中指定.h文件的位置。（在每个cpp文件中也可以逐一修改）。

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)

include_directories(${PROJECT_SOURCE_DIR}/include) #指定头文件目录

file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp) 


set(EXECUTABLE_OUTPUT_PATH /root/Cmaketest/build) #设置输出文件路径
add_executable(HelloWorld ${SRC})

制作库文件

在Cmake中

使用add_library([库名称] SHARED [所包含的源文件])来制作动态库

使用add_library([库名称] STATIC [所包含的源文件])来制作静态库

在linux中，静态库的后缀为.a 动态库的后缀为.so

设置库文件的生成路径

set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib)

使用set命令和LIBRARY_OUTPUT_PATH宏来指定输出路径。

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)

include_directories(${PROJECT_SOURCE_DIR}/include)

aux_source_directory(${PROJECT_SOURCE_DIR}/src SRC) 

# set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib) #设置可执行文件输出文件路径

set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib) #设置库文件输出文件路径
add_library(clac SHARED ${SRC})  #生成库文件

使用库文件

使用静态库

link_directories(${PROJECT_SOURCE_DIR}/lib) #指定库文件的路径
link_libraries(calc)  # 链接静态库

使用动态库

cmake_minimum_required(VERSION 3.10)
project(HelloWorld)

# 设置C++的标准版本为11
set(CMAKE_CXX_STANDARD 11)

# 包含头文件目录
include_directories(${PROJECT_SOURCE_DIR}/include)

# 设置源文件列表
set(SOURCE_FILES
    ${PROJECT_SOURCE_DIR}/src/add.cpp
    ${PROJECT_SOURCE_DIR}/src/div.cpp
    ${PROJECT_SOURCE_DIR}/src/mult.cpp
    ${PROJECT_SOURCE_DIR}/src/sub.cpp)

# 设置可执行文件输出文件路径
set(EXECUTABLE_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/build) 
#file(MAKE_DIRECTORY ${EXECUTABLE_OUTPUT_PATH}) 

# 设置库文件输出文件路径 
set(LIBRARY_OUTPUT_PATH ${PROJECT_SOURCE_DIR}/lib1)  
#file(MAKE_DIRECTORY ${LIBRARY_OUTPUT_PATH}) 

# 创建一个名为calc的共享库
add_library(calc SHARED ${SOURCE_FILES})

# 查找名为calc的库文件，并将路径设置为LIBRARY_OUTPUT_PATH
find_library(CALC_LIB calc PATHS ${LIBRARY_OUTPUT_PATH})
#link_directories(${CMAKE_CURRENT_SOURCE_DIR}/lib1)

# 创建一个名为HelloWorld的可执行程序
add_executable(HelloWorld ${PROJECT_SOURCE_DIR}/main.cpp)
# 将 CALC_LIB 链接到 Hello World 应用程序中
target_link_libraries(HelloWorld ${CALC_LIB})

上述代码中：

  1. 将设置动态库文件路径的命令由link_libraries换成了find_library。据说使用link_libraries会在某种情况下报错。
  2. 源文件目录使用逐一列举的方式进行，之前采用file(GLOB SRC ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)，该方式可能会导致添加新源文件后编译不识别的情况出现。
  3. 在实际使用中发现，上述代码在win平台下编译成功之后，会生成exe可执行文件，但是运行时会报错：找不到动态库文件；解决方法：将动态库ddl文件和exe文件放在同一目录下即可成功运行…；针对上述问题，目前还没有找到从根源解决的办法，待后续学习再做处理。
