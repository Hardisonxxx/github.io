def TiQuImgLianJei(filetxt):
    urls = []
    for line in filetxt:
        if 'img' in line:
            url = line.split('src=')[-1].split('"')[1]
            if 'http' in url:
                urls.append(url)
    return urls

def XianShiLianJei(TiQuDaoDeLianJei):
